@echo off
set currentPath=%cd%

if exist emsdk (
cd emsdk
git pull
cd..
) else (
git clone https://github.com/emscripten-core/emsdk.git
)

cd emsdk
emsdk_env.bat
emsdk install latest
emsdk activate latest

cd %currentPath%/vcpkg
bootstrap-vcpkg.bat
vcpkg install --triplet wasm32-emscripten

cmake -S . -B build-wasm -DVCPKG_TARGET_TRIPLET=wasm32-emscripten

Pause
